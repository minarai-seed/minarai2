// ーーー 設定 ーーー // GitHub Actionsによって、公開時に __GEMINI_API_KEY_PLACEHOLDER__ の部分が本物のキーに自動置換されます const GEMINI_API_KEY = "__GEMINI_API_KEY_PLACEHOLDER__"; const GEMINI_URL = `https://googleapis.com{GEMINI_API_KEY}`; // ページが開いた瞬間に、ユーザーにAPIキーを入力させるポップアップを出す window.addEventListener('DOMContentLoaded', () => { // 前回入力したキーがブラウザ（LocalStorage）に残っていれば自動で取得 const savedKey = localStorage.getItem('my_gemini_key'); const inputKey = prompt("【安全対策】Google AI StudioのAPIキーを入力してください。\n（このキーはあなた自身のブラウザ内のみに安全に保存され、ネット公開されません）:", savedKey || ""); if (inputKey) { GEMINI_API_KEY = inputKey; localStorage.setItem('my_gemini_key', inputKey); // 次回起動時のためにブラウザに保存 GEMINI_URL = `https://googleapis.com{GEMINI_API_KEY}`; } else { alert("APIキーが入力されなかったため、アプリは動作しません。ページを再読み込みして入力してください。"); document.getElementById('talk-btn').disabled = true; } });
